from [vchs-budget].dbo.mv_faculty_salaries_and_benefits as v1

select somDivisionId as divisionId,

from [vchs-staffing].dbo.V_ACAD_STAFFING_INDEX 
where FinManLineItemId in ('101','102')
AND fund in ('60105A','60109A')
